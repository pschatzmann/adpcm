155  inline uint8_t adpcm_ima_compress_sample(ADPCMChannelStatus *c,
156  int16_t sample) {
157  int delta = sample - c->prev_sample;
158  int nibble = FFMIN(7, abs(delta) * 4 / ff_adpcm_step_table[c->step_index]) +
159  (delta < 0) * 8;
160  c->prev_sample += ((ff_adpcm_step_table[c->step_index] *
161  ff_adpcm_yamaha_difflookup[nibble]) /
162  8);
163  c->prev_sample = av_clip_int16(c->prev_sample);
164  c->step_index =
165  av_clip(c->step_index + ff_adpcm_index_table[nibble], 0, 88);
166  return nibble;
167  }

169  inline uint8_t adpcm_ima_qt_compress_sample(ADPCMChannelStatus *c,
170  int16_t sample) {
171  int delta = sample - c->prev_sample;
172  int diff, step = ff_adpcm_step_table[c->step_index];
173  int nibble = 8 * (delta < 0);
174 
175  delta = abs(delta);
176  diff = delta + (step >> 3);
177 
178  if (delta >= step) {
179  nibble |= 4;
180  delta -= step;
181  }
182  step >>= 1;
183  if (delta >= step) {
184  nibble |= 2;
185  delta -= step;
186  }
187  step >>= 1;
188  if (delta >= step) {
189  nibble |= 1;
190  delta -= step;
191  }
192  diff -= delta;
193 
194  if (nibble & 8)
195  c->prev_sample -= diff;
196  else
197  c->prev_sample += diff;
198 
199  c->prev_sample = av_clip_int16(c->prev_sample);
200  c->step_index =
201  av_clip(c->step_index + ff_adpcm_index_table[nibble], 0, 88);
202 
203  return nibble;
204  }

206  inline uint8_t adpcm_yamaha_compress_sample(ADPCMChannelStatus *c,
207  int16_t sample) {
208  int nibble, delta;
209 
210  if (!c->step) {
211  c->predictor = 0;
212  c->step = 127;
213  }
214 
215  delta = sample - c->predictor;
216 
217  nibble = FFMIN(7, abs(delta) * 4 / c->step) + (delta < 0) * 8;
218 
219  c->predictor += ((c->step * ff_adpcm_yamaha_difflookup[nibble]) / 8);
220  c->predictor = av_clip_int16(c->predictor);
221  c->step = (c->step * ff_adpcm_yamaha_indexscale[nibble]) >> 8;
222  c->step = av_clip(c->step, 127, 24576);
223

260 class ADPCMEncoderTrellis : public ADPCMEncoder {
261  public:
262  bool is_trellis() { return avctx.trellis; }
263  void set_trellis(bool flag) { avctx.trellis = flag;}
264  bool store_node(int STEP_INDEX) {
265  int d;
266  uint32_t ssd;
267  int pos;
268  TrellisNode *u;
269  uint8_t *h;
270  dec_sample = av_clip_int16(dec_sample);
271  d = sample - dec_sample;
272  ssd = nodes[j]->ssd +
273  d * (unsigned)d; /* Check for wraparound, skip such samples
274  * completely. \
275  * Note, changing ssd to a 64 bit variable would be \
276  * simpler, avoiding this check, but it's slower on \
277  * x86 32 bit at the moment. */
278  if (ssd < nodes[j]->ssd) {
279  /* Collapse any two states with the same previous
280  * sample value. One could also distinguish states by step and by 2nd
281  * to last sample, but the effects of that are negligible.
282  * Since nodes in the previous generation are iterated through a heap,
283  * they're roughly ordered from better to worse, but not strictly ordered.
284  * Therefore, an earlier node with the same sample value is better in most
285  * cases (and thus the current is skipped), but not strictly
286  * in all cases. Only skipping samples where ssd >= ssd of the earlier
287  * node with the same sample gives slightly worse quality, though, for
288  * some reason. */
289  return true;
290  }
291  h = &hash[(uint16_t)dec_sample];
292  if (*h == generation) return true;
293  if (heap_pos < frontier) {
294  pos = heap_pos++;
295  } else { /* Try to replace one of the leaf nodes with the new \
296  * one, but try a different slot each time. */
297  pos = (frontier >> 1) + (heap_pos & ((frontier >> 1) - 1));
298  if (ssd > nodes_next[pos]->ssd) return true;
299  heap_pos++;
300  }
301  *h = generation;
302  u = nodes_next[pos];
303  if (!u) {
304  av_assert(pathn < FREEZE_INTERVAL << avctx.trellis);
305  u = t++;
306  nodes_next[pos] = u;
307  u->path = pathn++;
308  }
309  u->ssd = ssd;
310  u->step = STEP_INDEX;
311  u->sample2 = nodes[j]->sample1;
312  u->sample1 = dec_sample;
313  paths[u->path].nibble = nibble;
314  paths[u->path].prev =
315  nodes[j]->path; /* Sift the newly inserted node up in the heap to \
316  * restore the heap property. */
317  while (pos > 0) {
318  int parent = (pos - 1) >> 1;
319  if (nodes_next[parent]->ssd <= ssd) break;
320  FFSWAP(TrellisNode *, nodes_next[parent], nodes_next[pos]);
321  pos = parent;
322  }
323  return false;
324  }

326  void loop_nodes(int16_t STEP_TABLE, int STEP_INDEX) {
327  const int predictor = nodes[j]->sample1;
328  const int div = (sample - predictor) * 4 / STEP_TABLE;
329  int nmin = av_clip(div - range, -7, 6);
330  int nmax = av_clip(div + range, -6, 7);
331  if (nmin <= 0) nmin--; /* distinguish -0 from +0 */
332  if (nmax < 0) nmax--;
333  for (nidx = nmin; nidx <= nmax; nidx++) {
334  const int nibble = nidx < 0 ? 7 - nidx : nidx;
335  dec_sample =
336  predictor + (STEP_TABLE * ff_adpcm_yamaha_difflookup[nibble]) / 8;
337  store_node(STEP_INDEX);
338  }
339  }

341  void adpcm_compress_trellis(const int16_t *samples, uint8_t *dst,
342  ADPCMChannelStatus *c, int n, int stride) {
343  // FIXME 6% faster if frontier is a compile-time constant
344  s = (ADPCMEncodeContext *)avctx.priv_data;
345  frontier = 1 << avctx.trellis;
346  version = avctx.codec_id;
347  paths = s->paths, *p;
348  node_buf = s->node_buf;
349  nodep_buf = s->nodep_buf;
350  nodes = nodep_buf; // nodes[] is always sorted by .ssd
351  nodes_next = nodep_buf + frontier;
352  pathn = 0, froze = -1, i, j, k, generation = 0;
353  hash = s->trellis_hash;
354  memset(hash, 0xff, 65536 * sizeof(*hash));
355 
356  memset(nodep_buf, 0, 2 * frontier * sizeof(*nodep_buf));
357  nodes[0] = node_buf + frontier;
358  nodes[0]->ssd = 0;
359  nodes[0]->path = 0;
360  nodes[0]->step = c->step_index;
361  nodes[0]->sample1 = c->sample1;
362  nodes[0]->sample2 = c->sample2;
363  if (version == AV_CODEC_ID_ADPCM_IMA_WAV ||
364  version == AV_CODEC_ID_ADPCM_IMA_QT ||
365  version == AV_CODEC_ID_ADPCM_IMA_AMV ||
366  version == AV_CODEC_ID_ADPCM_SWF)
367  nodes[0]->sample1 = c->prev_sample;
368  if (version == AV_CODEC_ID_ADPCM_MS) nodes[0]->step = c->idelta;
369  if (version == AV_CODEC_ID_ADPCM_YAMAHA) {
370  if (c->step == 0) {
371  nodes[0]->step = 127;
372  nodes[0]->sample1 = 0;
373  } else {
374  nodes[0]->step = c->step;
375  nodes[0]->sample1 = c->predictor;
376  }
377  }
378 
379  for (i = 0; i < n; i++) {
380  t = node_buf + frontier * (i & 1);
381  sample = samples[i * stride];
382  heap_pos = 0;
383  memset(nodes_next, 0, frontier * sizeof(TrellisNode *));
384  for (j = 0; j < frontier && nodes[j]; j++) {
385  // higher j have higher ssd already, so they're likely
386  // to yield a suboptimal next sample too
387  range = (j < frontier / 2) ? 1 : 0;
388  const int step = nodes[j]->step;
389  if (version == AV_CODEC_ID_ADPCM_MS) {
390  const int predictor = ((nodes[j]->sample1 * c->coeff1) +
391  (nodes[j]->sample2 * c->coeff2)) /
392  64;
393  const int div = (sample - predictor) / step;
394  const int nmin = av_clip(div - range, -8, 6);
395  const int nmax = av_clip(div + range, -7, 7);
396  for (nidx = nmin; nidx <= nmax; nidx++) {
397  nibble = nidx & 0xf;
398  dec_sample = predictor + nidx * step;
399 
400  while (store_node(
401  FFMAX(16, (ff_adpcm_AdaptationTable[nibble] * step) >> 8)));
402  }
403  } else if (version == AV_CODEC_ID_ADPCM_IMA_WAV ||
404  version == AV_CODEC_ID_ADPCM_IMA_QT ||
405  version == AV_CODEC_ID_ADPCM_IMA_AMV ||
406  version == AV_CODEC_ID_ADPCM_SWF) {
407  loop_nodes(ff_adpcm_step_table[step],
408  av_clip(step + ff_adpcm_index_table[nibble], 0, 88));
409  } else { // AV_CODEC_ID_ADPCM_YAMAHA
410  loop_nodes(step,
411  av_clip((step * ff_adpcm_yamaha_indexscale[nibble]) >> 8,
412  127, 24576));
413  }
414  }
415 
416  u = nodes;
417  nodes = nodes_next;
418  nodes_next = u;
419 
420  generation++;
421  if (generation == 255) {
422  memset(hash, 0xff, 65536 * sizeof(*hash));
423  generation = 0;
424  }
425 
426  // prevent overflow
427  if (nodes[0]->ssd > (1 << 28)) {
428  for (j = 1; j < frontier && nodes[j]; j++)
429  nodes[j]->ssd -= nodes[0]->ssd;
430  nodes[0]->ssd = 0;
431  }
432 
433  // merge old paths to save memory
434  if (i == froze + FREEZE_INTERVAL) {
435  p = &paths[nodes[0]->path];
436  for (k = i; k > froze; k--) {
437  dst[k] = p->nibble;
438  p = &paths[p->prev];
439  }
440  froze = i;
441  pathn = 0;
442  // other nodes might use paths that don't coincide with the frozen one.
443  // checking which nodes do so is too slow, so just kill them all.
444  // this also slightly improves quality, but I don't know why.
445  memset(nodes + 1, 0, (frontier - 1) * sizeof(TrellisNode *));
446  }
447  }
448 
449  p = &paths[nodes[0]->path];
450  for (i = n - 1; i > froze; i--) {
451  dst[i] = p->nibble;
452  p = &paths[p->prev];
453  }
454 
455  c->predictor = nodes[0]->sample1;
456  c->sample1 = nodes[0]->sample1;
457  c->sample2 = nodes[0]->sample2;
458  c->step_index = nodes[0]->step;
459  c->step = nodes[0]->step;
460  c->idelta = nodes[0]->step;
461  }
462 
463  protected:
464  int frontier;
465  int version;
466  TrellisPath *paths, *p;
467  TrellisNode *node_buf;
468  TrellisNode **nodep_buf;
469  TrellisNode **nodes; // nodes[] is always sorted by .ssd
470  TrellisNode **nodes_next;
471  TrellisNode **u;
472  TrellisNode *t;
473 
474  int pathn = 0, froze = -1, i, j, k, generation = 0;
475  uint8_t *hash;
476  int dec_sample = 0;
477  int sample;
478  int nibble;
479  int nidx;
480  int range;
481  int heap_pos;

501  int *got_packet_ptr) {
502  int blocks = (frame->nb_samples - 1) / 8;
503 
504  for (int ch = 0; ch < channels; ch++) {
505  ADPCMChannelStatus *status = &c->status[ch];
506  status->prev_sample = samples_p[ch][0];
507  /* status->step_index = 0;
508  XXX: not sure how to init the state machine */
509  bytestream_put_le16(&dst, status->prev_sample);
510  *dst++ = status->step_index;
511  *dst++ = 0; /* unknown */
512  }
513 
514  /* stereo: 4 bytes (8 samples) for left, 4 bytes for right */
515  if (avctx.trellis > 0) {
516  uint8_t *buf;
517  if (!FF_ALLOC_TYPED_ARRAY(uint8_t *, buf, channels *blocks * 8))
518  return AVERROR(ENOMEM);
519  for (int ch = 0; ch < channels; ch++) {
520  adpcm_compress_trellis(&samples_p[ch][1], buf + ch * blocks * 8,
521  &c->status[ch], blocks * 8, 1);
522  }
523  for (int i = 0; i < blocks; i++) {
524  for (int ch = 0; ch < channels; ch++) {
525  uint8_t *buf1 = buf + ch * blocks * 8 + i * 8;
526  for (int j = 0; j < 8; j += 2) *dst++ = buf1[j] | (buf1[j + 1] << 4);
527  }
528  }
529  av_free(buf);
530  } else {
531  for (int i = 0; i < blocks; i++) {
532  for (int ch = 0; ch < channels; ch++) {
533  ADPCMChannelStatus *status = &c->status[ch];
534  const int16_t *smp = &samples_p[ch][1 + i * 8];
535  for (int j = 0; j < 8; j += 2) {
536  uint8_t v = adpcm_ima_compress_sample(status, smp[j]);
537  v |= adpcm_ima_compress_sample(status, smp[j + 1]) << 4;
538  *dst++ = v;
539  }
540  }
541  }
542  }
543  return AV_OK;
544  }

694  bytestream_put_le16(&extradata, ff_adpcm_AdaptCoeff1[i] * 4);

700  inline uint8_t adpcm_ms_compress_sample(ADPCMChannelStatus *c,
701  int16_t sample) {
702  int predictor, nibble, bias;
703 
704  predictor =
705  (((c->sample1) * (c->coeff1)) + ((c->sample2) * (c->coeff2))) / 64;
706 
707  nibble = sample - predictor;
708  if (nibble >= 0)
709  bias = c->idelta / 2;
710  else
711  bias = -c->idelta / 2;
712 
713  nibble = (nibble + bias) / c->idelta;
714  nibble = av_clip_intp2(nibble, 3) & 0x0F;
715 
716  predictor += ((nibble & 0x08) ? (nibble - 0x10) : nibble) * c->idelta;
717 
718  c->sample2 = c->sample1;
719  c->sample1 = av_clip_int16(predictor);
720 
721  c->idelta = (ff_adpcm_AdaptationTable[nibble] * c->idelta) >> 8;
722  if (c->idelta < 16) c->idelta = 16;
723 
724  return nibble;
725  }

727  int adpcm_encode_frame_impl(AVPacket *avpkt, const AVFrame *frame,
728  int *got_packet_ptr) {
729  for (int i = 0; i < channels; i++) {
730  int predictor = 0;
731  *dst++ = predictor;
732  c->status[i].coeff1 = ff_adpcm_AdaptCoeff1[predictor];
733  c->status[i].coeff2 = ff_adpcm_AdaptCoeff2[predictor];
734  }
735  for (int i = 0; i < channels; i++) {
736  if (c->status[i].idelta < 16) c->status[i].idelta = 16;
737  bytestream_put_le16(&dst, c->status[i].idelta);
738  }
739  for (int i = 0; i < channels; i++) c->status[i].sample2 = *samples++;
740  for (int i = 0; i < channels; i++) {
741  c->status[i].sample1 = *samples++;
742  bytestream_put_le16(&dst, c->status[i].sample1);
743  }
744  for (int i = 0; i < channels; i++)
745  bytestream_put_le16(&dst, c->status[i].sample2);
746 
747  if (avctx.trellis > 0) {
748  const int n = avctx.block_align - 7 * channels;
749  uint8_t *buf = (uint8_t *)av_malloc(2 * n);
750  if (!buf) return AVERROR(ENOMEM);
751  if (channels == 1) {
752  adpcm_compress_trellis(samples, buf, &c->status[0], n, channels);
753  for (int i = 0; i < n; i += 2) *dst++ = (buf[i] << 4) | buf[i + 1];
754  } else {
755  adpcm_compress_trellis(samples, buf, &c->status[0], n, channels);
756  adpcm_compress_trellis(samples + 1, buf + n, &c->status[1], n,
757  channels);
758  for (int i = 0; i < n; i++) *dst++ = (buf[i] << 4) | buf[n + i];
759  }
760  av_free(buf);
761  } else {
762  for (int i = 7 * channels; i < avctx.block_align; i++) {
763  int nibble;
764  nibble = adpcm_ms_compress_sample(&c->status[0], *samples++) << 4;
765  nibble |= adpcm_ms_compress_sample(&c->status[st], *samples++);
766  *dst++ = nibble;
767  }
768  } /* End of CASE */
769  return AV_OK;
770  }

1010  else

1021  int64_t adpcm_argo_compress_block(ADPCMChannelStatus *cs, PutBitContext *pb,
1022  const int16_t *samples, int nsamples,
1023  int shift, int flag) {
1024  int64_t error = 0;
1025 
1026  if (pb) {
1027  put_bits(pb, 4, shift - 2);
1028  put_bits(pb, 1, 0);
1029  put_bits(pb, 1, !!flag);
1030  put_bits(pb, 2, 0);
1031  }
1032 
1033  for (int n = 0; n < nsamples; n++) {
1034  /* Compress the nibble, then expand it to see how much precision we've
1035  * lost. */
1036  int nibble = adpcm_argo_compress_nibble(cs, samples[n], shift, flag);
1037  int16_t sample = ff_adpcm_argo_expand_nibble(cs, nibble, shift, flag);
1038 
1039  error += abs(samples[n] - sample);
1040 
1041  if (pb) put_bits(pb, 4, nibble);
1042  }
1043 
1044  return error;
1045  }

1051

1053

1056  int shift = 2, flag = 0;
1057  int saved1 = c->status[ch].sample1;
1058  int saved2 = c->status[ch].sample2;
1059 
1060  /* Find the optimal coefficients, bail early if we find a perfect
1061  * result. */
1062  for (int s = 2; s < 18 && tmperr != 0; s++) {
1063  for (int f = 0; f < 2 && tmperr != 0; f++) {

1067  c->status + ch, NULL, samples_p[ch], frame->nb_samples, s, f);
1068  if (tmperr < error) {
1069  shift = s;
1070  flag = f;
1071  error = tmperr;
1072  }
1073  }
1074  }
1075 
1076  /* Now actually do the encode. */
1077  c->status[ch].sample1 = saved1;
1078  c->status[ch].sample2 = saved2;
